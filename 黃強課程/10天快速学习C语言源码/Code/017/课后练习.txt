/*
编写一个float数组存储3个班级，每班10个学生的成绩，
并求出每班的平均成绩！
 */
#include <stdio.h> 

void   main()
{ 
	//定义的时候先初始化8个学生成绩
	float  scores[3][10] = {
		 {100,62,36,42,80,64,99,90},
		 { 60,85,36,32,77,72,96,72 },
	     { 90,64,86,59,87,99,69,55 }
	};
 

	float sum[3] = { 0 }; 
	for (int i = 0; i < 3; ++i)
	 for (int j = 0; j < 10; ++j)
	 {
			sum[i] += scores[i][j];
	 } 
	
	for (int i = 0; i < 3; ++i)
	{
		printf("班级%d学生的平均成绩为%.2f\n",i+1,  sum[i] / 10);
	 }
	
}
