/*
练习如下程序，并说清楚程序为什么会输出这些结果！
 */
#include <stdio.h> 

int  a = 100; //全局变量a

void  fun()
{
    a = 200; //这里访问的是全局变量，将全局变量的值改为200
    printf("fun中的a是%d \n", a);
}

void   main()
{
    printf("main中的a是%d \n", a); //这里访问的是全局变量100
    int  a = 300; //这里是局部变量
    fun();//这里将全局变量的值改为200
    printf("main中的a是%d \n", a);//这里访问的是离它最近的局部变量a ，屏蔽了外面的全局变量，打印300
}
